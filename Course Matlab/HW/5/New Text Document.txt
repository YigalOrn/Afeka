function
a=liketrapz1(x,y)
dx=x(2)-x(1);
for
i=1:(length(x)-1)
area(i)=0.5*(y(i)+y(i+1))*dx;
end
a=sum(area);
 
 
function
a=liketrapz2(x,y)
dx=x(2)-x(1);
a=(sum(y)-0.5*y(1)-0.5*y(length(y)))*dx;
